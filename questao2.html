num_g = [K2]

# 2. Clculo do fator de amortecimento para 20% de sobressinal
Mp_percent = 20.0
Mp_decimal = Mp_percent / 100.0
zeta_d = (-np.log(Mp_decimal)) / np.sqrt(np.pi**2 + (np.log(Mp_decimal))**2)
print(f"\n2. ESPECIFICAES:")
print(f"Sobressinal desejado: {Mp_percent}%")
print(f"Fator de amortecimento:  = {zeta_d:.3f}")

# 3. ngulo da linha de amortecimento constante
theta = np.arccos(zeta_d)
theta_deg = np.degrees(theta)
print(f"ngulo da linha de amortecimento:  = {theta_deg:.1f}")

# 4. Plo dominante desejado para tempo de acomodao 4x menor
sigma_desired = 4 / ts_c2  # ts_c2 foi calculado como ts_nc2/4
wd_desired = sigma_desired * np.tan(theta)
s_desired = -sigma_desired + 1j*wd_desired

print(f"\n3. PLO DOMINANTE DESEJADO:")
print(f"_d = {sigma_desired:.2f} rad/s")
print(f"_d = {wd_desired:.2f} rad/s")
print(f"s_d = {s_desired:.2f}")

# 5. Verificao da condio de ngulo no ponto desejado
# Para G(s) = K/[s(s+5)(s+15)], temos plos em s=0, s=-5, s=-15
s_val = s_desired

# ngulos dos plos existentes para o ponto desejado
angle_s0 = np.angle(s_val - 0)  # plo na origem
angle_s5 = np.angle(s_val - (-5))  # plo em -5
angle_s15 = np.angle(s_val - (-15))  # plo em -15

# Soma dos ngulos existentes
soma_angulos_existentes = angle_s0 + angle_s5 + angle_s15
soma_angulos_existentes_deg = np.degrees(soma_angulos_existentes)

print(f"\n4. ANLISE DE NGULOS NO PONTO DESEJADO:")
print(f"ngulo de s: {np.degrees(angle_s0):.1f}")
print(f"ngulo de (s+5): {np.degrees(angle_s5):.1f}")
print(f"ngulo de (s+15): {np.degrees(angle_s15):.1f}")
print(f"Soma dos ngulos: {soma_angulos_existentes_deg:.1f}")

# 6. ngulo necessrio do compensador PD
# Condio de ngulo: soma total deve ser -180 (ou mltiplo mpar de 180)
angulo_necessario = -180 - soma_angulos_existentes_deg
while angulo_necessario < -180:
    angulo_necessario += 360
while angulo_necessario > 180:
    angulo_necessario -= 360

print(f"ngulo necessrio do compensador: {angulo_necessario:.1f}")

# 7. Projeto do compensador PD: Gc(s) = Kp + Kd*s
# O compensador PD contribui com um zero, precisamos calcular sua localizao

if abs(angulo_necessario) > 5:  # Se precisamos de compensao significativa
    # Para compensador PD: Gc(s) = Kd(s + z) onde z = Kp/Kd
    # O zero deve ser colocado para fornecer o ngulo necessrio
    
    # Calculando a localizao do zero para atingir o ngulo necessrio
    # Usando geometria do lugar das razes
    angulo_zero_rad = np.radians(angulo_necessario)
    
    # Para um zero em -z, o ngulo contribudo  arctan(Im(s)/(Re(s) + z))
    # angulo_zero = arctan(wd_desired / (sigma_desired - z))
    # z = sigma_desired - wd_desired / tan(angulo_zero)
    
    if abs(np.tan(angulo_zero_rad)) > 1e-6:
        z_compensador = sigma_desired - wd_desired / np.tan(angulo_zero_rad)
    else:
        z_compensador = sigma_desired  # Zero na mesma posio da parte real
    
    print(f"\n5. COMPENSADOR PD:")
    print(f"Zero do compensador: z = {z_compensador:.3f}")
    
    # Se z_compensador > 0, temos Kp e Kd vlidos
    if z_compensador > 0:
        # Gc(s) = Kd(s + z) = Kd*s + Kd*z = Kd*s + Kp
        # Portanto: Kp = Kd * z
        # Escolhemos Kd = 1 para simplicidade, ento Kp = z
        Kd_calc = 1.0
        Kp_calc = z_compensador
    else:
        print("AVISO: Zero calculado  negativo, usando abordagem alternativa")
        # Usar mtodo direto baseado na condio de mdulo
        Kp_calc = 1.0
        Kd_calc = 0.1
        
else:
    print("\n5. COMPENSADOR SIMPLES:")
    print("ngulo j est prximo de -180, compensador simples")
    Kp_calc = 1.0
    Kd_calc = 0.1

print(f"Kp = {Kp_calc:.3f}")
print(f"Kd = {Kd_calc:.3f}")
print(f"Gc(s) = {Kp_calc:.3f} + {Kd_calc:.3f}s")

# 8. Verificao usando condio de mdulo
# |Gc(s_d) * G(s_d)| = 1
modulo_G = abs(K2 / (s_val * (s_val + 5) * (s_val + 15)))
modulo_Gc = abs(Kp_calc + Kd_calc * s_val)
modulo_total = modulo_G * modulo_Gc

print(f"\n6. VERIFICAO DA CONDIO DE MDULO:")
print(f"|G(s_d)| = {modulo_G:.6f}")
print(f"|Gc(s_d)| = {modulo_Gc:.6f}")
print(f"|Gc(s_d) * G(s_d)| = {modulo_total:.6f}")
print(f"Deve ser prximo de 1 para o ponto estar no lugar das razes")

# Ajustar ganho para condio de mdulo se necessrio
if modulo_total > 0:
    fator_ajuste = 1.0 / modulo_total
    K2_ajustado = K2 * fator_ajuste
    print(f"Ganho ajustado: K = {K2_ajustado:.1f}")
else:
    K2_ajustado = K2

def analyze_step_response(t, y):
    """Analisa caractersticas da resposta ao degrau"""
    yss = y[-1]
    
    # Tempo de pico
    peak_idx = np.argmax(y)
    tp = t[peak_idx]
    peak_value = y[peak_idx]
    
    # Sobressinal percentual
    Mp = (peak_value - yss) / yss * 100
    
    # Tempo de acomodao (critrio de 2%)
    settling_band = 0.02 * yss
    settled_idx = np.where(np.abs(y - yss) <= settling_band)[0]
    ts = t[settled_idx[0]] if len(settled_idx) > 0 else t[-1]
    
    # Tempo de subida (10% a 90%)
    y10, y90 = 0.1 * yss, 0.9 * yss

# Anlise de margem de fase e ganho
def analyze_margins_alternative(sys_ol, label):
    """Analisa margens de estabilidade usando resposta em frequncia"""
    try:
        # Calcular resposta em frequncia
        w = np.logspace(-2, 3, 1000)
        w_resp, mag_resp, phase_resp = signal.bode(sys_ol, w)
        
        # Converter magnitude para escala linear se necessrio
        mag_linear = 10**(mag_resp/20)
        
        # Encontrar frequncia de cruzamento de ganho (|H(jw)| = 1)
        unity_gain_idx = np.argmin(np.abs(mag_linear - 1))
        wg = w_resp[unity_gain_idx] if unity_gain_idx < len(w_resp) else np.nan
        
        # Margem de fase no cruzamento de ganho
        if not np.isnan(wg):
            phase_at_unity = phase_resp[unity_gain_idx]
            pm = 180 + phase_at_unity  # Margem de fase
        else:
            pm = np.nan
            
        # Encontrar frequncia de cruzamento de fase (-180)
        phase_cross_idx = np.argmin(np.abs(phase_resp + 180))
        wp = w_resp[phase_cross_idx] if phase_cross_idx < len(w_resp) else np.nan
        
        # Margem de ganho no cruzamento de fase
        if not np.isnan(wp):
            mag_at_phase_cross = mag_resp[phase_cross_idx]
            gm_db = -mag_at_phase_cross  # Margem de ganho em dB
        else:
            gm_db = np.inf
            
        print(f"\n{label}:")
        if gm_db == np.inf:
            print(f"  Margem de Ganho:  dB (sistema estvel)")
        else:
            print(f"  Margem de Ganho: {gm_db:.1f} dB")
        
        if not np.isnan(pm):
            print(f"  Margem de Fase: {pm:.1f}")
        else:
            print(f"  Margem de Fase: N/A")
            
        if not np.isnan(wg):
            print(f"  Freq. de cruzamento de ganho: {wg:.2f} rad/s")
        if not np.isnan(wp):
            print(f"  Freq. de cruzamento de fase: {wp:.2f} rad/s")
            
    except Exception as e:
        print(f"\n{label}: Erro no clculo das margens")
        print(f"  Erro: {str(e)}")
